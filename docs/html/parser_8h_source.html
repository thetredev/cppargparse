156  const std::string usage() const
157  {
158  // Find the maximum (argument ID, argument alternative ID) length
159  size_t max_id_length = 0;
160 
161  for (const auto &cmdarg : m_cmdargs)
162  {
163  const size_t id_length = cmdarg.id.size() + cmdarg.id_alt.size();
164 
165  if (id_length > max_id_length)
166  {
167  max_id_length = id_length;
168  }
169  }
170 
171  // Add 3 more spaces
172  max_id_length += 3;
173 
174 
175  // Generate the usage string
176  std::ostringstream usage_string;
177  usage_string << "Usage: " << m_application_description << '\n' << '\n';
178 
179  for (const auto &cmdarg : m_cmdargs)
180  {
181  usage_string << ' ' << ' ' << cmdarg.id;
182 
183  if (!cmdarg.id_alt.empty())
184  {
185  usage_string << '|' << cmdarg.id_alt;
186  }
187 
188  for (size_t i = 0; i < max_id_length; ++i)
189  {
190  usage_string << ' ';
191  }
192 
193  if (!cmdarg.description.empty())
194  {
195  usage_string << cmdarg.description;
196  }
197 
198  usage_string << '\n';
199  }
200 
201  return usage_string.str();
202  }
