173  const std::string usage() const
174  {
175  // Find the maximum (argument ID, argument alternative ID) length
176  size_t max_id_length = 0;
177 
178  for (const auto &cmdarg : m_cmdargs)
179  {
180  const size_t id_length = cmdarg.id.size() + cmdarg.id_alt.size();
181 
182  if (id_length > max_id_length)
183  {
184  max_id_length = id_length;
185  }
186  }
187 
188  // Add 3 more spaces
189  max_id_length += 3;
190 
191 
192  // Generate the usage string
193  std::ostringstream usage_string;
194  usage_string << "Usage: " << m_application_description << '\n' << '\n';
195 
196  for (const auto &cmdarg : m_cmdargs)
197  {
198  usage_string << ' ' << ' ' << cmdarg.id;
199 
200  if (!cmdarg.id_alt.empty())
201  {
202  usage_string << '|' << cmdarg.id_alt;
203  }
204 
205  for (size_t i = 0; i < max_id_length; ++i)
206  {
207  usage_string << ' ';
208  }
209 
210  if (!cmdarg.description.empty())
211  {
212  usage_string << cmdarg.description;
213  }
214 
215  usage_string << '\n';
216  }
217 
218  return usage_string.str();
219  }
